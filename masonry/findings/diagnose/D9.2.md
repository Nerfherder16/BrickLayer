```javascript
function atomicWrite(p, obj) {
  const tmp = `${p}.tmp.${process.pid}`;
  try {
    fs.writeFileSync(tmp, JSON.stringify(obj, null, 2), "utf8");
    fs.renameSync(tmp, p);
  } catch (_) {
    try { fs.unlinkSync(tmp); } catch (_2) {}
  }
}
```
