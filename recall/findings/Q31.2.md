```python
def _user_conditions(self, user_id: int | None) -> list:
    if user_id is not None and user_id > 0:
        return [FieldCondition(key="user_id", match=MatchValue(value=user_id))]
    if user_id == 0:
        return [IsNullCondition(is_null=PayloadField(key="user_id"))]
    return []  # user_id=None → no filter
```
