class GraphitiStore:
    """Temporal knowledge graph store via Graphiti."""

    def __init__(self):
        self.settings = get_settings()
        self._graphiti: Any | None = None
        self._initialized = False

    async def initialize(self) -> bool:
        """Initialize Graphiti with Ollama LLM and embedder."""
        if self._initialized:
            return True

        if not self.settings.graphiti_enabled:
            logger.info("graphiti_disabled")
            return False

        try:
            from graphiti_core import Graphiti
            from graphiti_core.embedder.openai import OpenAIEmbedder, OpenAIEmbedderConfig
            from graphiti_core.llm_client.config import LLMConfig
            from graphiti_core.llm_client.openai_client import OpenAIClient

            ollama_base_url = f"{self.settings.ollama_host}/v1"

            llm = OpenAIClient(
                config=LLMConfig(
                    api_key="ollama",  # Ollama does not need a real key
                    base_url=ollama_base_url,
                    model=self.settings.graphiti_llm_model,
                )
            )

            embedder = OpenAIEmbedder(
                config=OpenAIEmbedderConfig(
                    api_key="ollama",
                    base_url=ollama_base_url,
                    embedding_model=self.settings.embedding_model,
                )
            )

            self._graphiti = Graphiti(
                uri=self.settings.neo4j_uri,
                user=self.settings.neo4j_user,
                password=self.settings.neo4j_password,
                llm_client=llm,
                embedder=embedder,
            )

            # Build indexes (idempotent -- safe on existing DB)
            await self._graphiti.build_indices_and_constraints()
            self._initialized = True
            logger.info("graphiti_initialized")
            return True

        except Exception as e:
            logger.error("graphiti_init_failed", error=str(e))
            return False
