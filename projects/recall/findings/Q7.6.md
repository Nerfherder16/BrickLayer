"""
Hygiene worker - auto-archives stale, low-value memories.

Runs daily at 4:00 AM. Soft-deletes memories where:
- access_count == 0 AND importance < 0.3 AND age > 30 days
- NOT pinned, NOT permanent durability
"""

from datetime import UTC, datetime, timedelta

import structlog

from src.storage import get_postgres_store, get_qdrant_store

logger = structlog.get_logger()


async def run_hygiene(ctx: dict):
    """Auto-archive stale, low-value memories."""
    logger.info("running_hygiene")

    qdrant = await get_qdrant_store()
    pg = await get_postgres_store()
    cutoff = datetime.now(UTC) - timedelta(days=30)
    cutoff_iso = cutoff.isoformat()

    archived = 0
    # Server-side filtered: only candidates that pass all hygiene criteria
    candidates = await qdrant.scroll_hygiene_candidates(cutoff_iso)

    for memory_id, payload in candidates:
        importance = payload.get("importance", 0.5)
        access_count = payload.get("access_count", 0)

        # Compute age for the audit log
        created_str = payload.get("created_at", "")
        try:
            created_at = datetime.fromisoformat(created_str.replace("Z", "+00:00"))
            age_days = (datetime.now(UTC) - created_at).days
        except (ValueError, TypeError):
            age_days = 0

        # Archive: mark as superseded
        try:
            await qdrant.update_payload(
                memory_id,
                {"superseded_by": "auto-archive"},
            )
            await pg.log_audit(
                action="auto_archive",
                memory_id=memory_id,
                actor="hygiene",
                details={
                    "importance": importance,
                    "access_count": access_count,
                    "age_days": age_days,
                },
            )
            archived += 1
        except Exception as e:
            logger.warning("hygiene_archive_failed", id=memory_id, error=str(e))

    logger.info("hygiene_complete", archived=archived)
    return {"archived": archived}
